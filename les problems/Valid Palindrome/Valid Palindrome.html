	def isAnagram(self, s, t):
		d={}
        q={}
        for i in (s):
            if i in d:
                d[i]=d[i]+1
            else :
                d[i]=1
            
        for i in (t):
            if i in q:
                q[i]=q[i]+1
            else :
                q[i]=1
        if len(d)>len(q):
            for i in (d):
                if i not in q:
                    return False
                if d[i]!=q[i]:
                    return False
        else:
            for i in (q):
                if i not in d:
                    return False
                if d[i]!=q[i]:
                    return False
        return True
