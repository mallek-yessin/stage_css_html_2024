Longest Increasing Subsequence

https://codeforces.com/problemset/gymProblem/102961/G

time complexity O(n^2)


def lengthOfLIS(self, nums):
    # Création d'une liste l avec des valeurs initiales de 1, une pour chaque élément de nums
    l = [1] * len(nums)

    # Parcours des éléments de nums à partir de l'indice 1 jusqu'à la fin
    for i in range(1, len(nums)):
        # Parcours des éléments de nums jusqu'à l'indice i
        for j in range(0, i):
            # Vérification si l'élément à l'indice i est plus grand que l'élément à l'indice j
            # et si l'élément à l'indice i forme une sous-séquence croissante plus longue que celle
            # formée par l'élément à l'indice j
            if nums[i] > nums[j] and l[i] < l[j] + 1:
                # Si oui, met à jour l[i] avec l[j] + 1, car cela signifie qu'on peut prolonger la sous-séquence
                l[i] = l[j] + 1

    # Initialisation d'une variable ma à 0 qui stockera la longueur maximale de la sous-séquence croissante
    ma = 0
    # Parcours des éléments de l
    for i in range(len(nums)):
        # Mise à jour de ma avec la valeur maximale entre ma et l[i]
        ma = max(ma, l[i])

    # Retourne la longueur maximale de la sous-séquence croissante
    return ma
