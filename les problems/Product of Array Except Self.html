Product of Array Except Self

https://codeforces.com/problemset/gymProblem/102961/G

on a calcule le produit de la liste nums

dans ce programme il y a un cas particulier ("0" dans nums)

si il y a 2 ou plusieurs 0 tout les valeurs de l sont 0

il y a une seule 0 tout les valeurs de l sont 0 sauf la valeur de l'indice de la valeur 0 dans nums est le produit dans l

time complexity O(n)


def productExceptSelf(self, nums: List[int]) -> List[int]:
    # Initialiser une liste vide pour stocker les résultats.
    l = []
    # Vérifier si zéro est présent dans la liste nums.
    a = 0 in nums
    # Initialiser une variable pour le produit total.
    p = 1
    # Initialiser un drapeau pour suivre si un zéro a été rencontré.
    t = False
    
    # Calculer le produit de tous les éléments de la liste nums, sauf zéro.
    for i in nums:
        # Si l'élément est non nul ou si un zéro a déjà été rencontré.
        if i != 0 or t:
            p = p * i
        else:
            # Sinon, marquer qu'un zéro a été rencontré.
            t = True
    
    # Si aucun zéro n'est présent dans la liste nums.
    if not a:
        # Parcourir la liste nums.
        for i in range(len(nums)):
            # Si l'élément n'est pas zéro, calculer le produit total divisé par l'élément.
            if nums[i] != 0:
                l.append(p // nums[i])
            else:
                # Si l'élément est zéro, ajouter le produit total à la liste résultante.
                l.append(p)
        # Retourner la liste de produits.
        return l
    else:
        # Si zéro est présent dans la liste nums.
        # Parcourir la liste nums.
        for i in range(len(nums)):
            # Si l'élément n'est pas zéro, ajouter zéro à la liste résultante.
            if nums[i] != 0:
                l.append(0)
            else:
                # Si l'élément est zéro, ajouter le produit total à la liste résultante.
                l.append(p)
        # Retourner la liste de produits.
        return l
