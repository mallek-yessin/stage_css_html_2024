    def missingNumber(self, nums: List[int]) -> int:
        n = len(nums)
        ans = 0
        for i in range(1, n + 1):
            ans ^= i
        for num in nums:
            ans ^= num
        return ans

    def missingNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        s=set()
        for i in  (nums):
            s.add(i)
        
        for i in range (len(nums)):
