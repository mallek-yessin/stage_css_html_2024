  def mergeSort(arr):
      if len(arr) <= 1:
          return arr
  
      mid = len(arr) // 2
      leftHalf = arr[:mid]
      rightHalf = arr[mid:]
  
      sortedLeft = mergeSort(leftHalf)
      sortedRight = mergeSort(rightHalf)
  
      return merge(sortedLeft, sortedRight)
  
  def merge(left, right):
      result = []
      i = j = 0
  
      while i < len(left) and j < len(right):
          if left[i] < right[j]:
              result.append(left[i])
              i += 1
          else:
              result.append(right[j])
              j += 1
  
      result.extend(left[i:])
      result.extend(right[j:])
  
      return result
  
  unsortedArr = [3, 7, 6, -10, 15, 23.5, 55, -13]
  sortedArr = mergeSort(unsortedArr)
  print("Sorted array:", sortedArr)

          def fast_power_recursive(x, y, m):
          # Calculate (x ^ y) % m using recursion
          if y == 0:
              return 1
          p = fast_power_recursive(x, y // 2, m)  # half the exponent
          ans = (p * p) % m
          if y % 2 == 1:  # if exponent is odd
              ans = (ans * x) % m
          return ans
      
      def fast_power_loop(x, y, m):
          # Calculate (x ^ y) % m using a loop
          ans = 1
          p = x  # x raised to powers of 2 (1, 2, 4, 8, ...)
          while y > 0:
              if y % 2 == 1:
                  ans = (ans * p) % m
              p = (p * p) % m
              y //= 2  # divide y by 2
          return ans
      
      # Main part to test the functions

    def phi(m):
    """Computes Euler's Totient function Ï†(m)."""
    result = m
    p = 2
    while p * p <= m:
        # Check if p divides m
        if m % p == 0:
            # Divide out all occurrences of p from m
            while m % p == 0:
                m //= p
            result -= result // p
        p += 1
    # If m is still greater than 1, it's prime
    if m > 1:
        result -= result // m
    return result


def mod_exp_with_phi(a, x, m):
    phi_m = phi(m)
    # Reduce exponent modulo phi(m)
    reduced_exponent = x % phi_m
    return mod_exp(a, reduced_exponent, m)

def mod_exp(a, x, m):
    result = 1
    a = a % m
    while x > 0:
        if x % 2 == 1:
            result = (result * a) % m
        x = x // 2
        a = (a * a) % m
    return result
